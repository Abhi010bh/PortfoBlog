package com.mycompany.tw1;
import java.io.*;
import java.io.IOException;
import java.util.Scanner;
import static java.lang.System.*;



/**
 *
 * @author abhib
 */
public class TW1 {

    public static void printCharacter(File srcFile){
        
       try{ 
           BufferedReader reader=new BufferedReader(new FileReader(srcFile));
           
           int c;
           while((c=reader.read())!=-1){
               out.println((char)c);
           }
           
           reader.close();
       }
       catch(IOException e){
           e.printStackTrace();
       }
       
    }
    
    public static void printFile(File srcFile){
        
        try{
            BufferedReader reader=new BufferedReader(new FileReader(srcFile));
            
            String line;
            while((line=reader.readLine())!=null){
            out.println(line);
            }
            
            reader.close();
        }
        catch(IOException e){
            e.printStackTrace();
        }
    }
    
    public static void copyFileContent(File srcFile){
        Scanner Sc=new Scanner(System.in);
    
        try{
            BufferedReader reader=new BufferedReader(new FileReader(srcFile));
            
            out.println("Enter destination file name");
            String desFileName=Sc.nextLine();
            
            File desFile=new File(desFileName+".txt");
            if(desFile.exists()){
                out.println("Overwriting targetFile");
            }

            BufferedWriter writer=new BufferedWriter(new FileWriter(desFile));
            String line;
            
            while((line=reader.readLine())!=null){
                writer.write(line);
            }
            reader.close();
            writer.close();
            
        }
        catch(IOException e){
            e.printStackTrace();
        }
    }
    
    public static void main(String[] args) {
        //create a file object 
        Scanner Sc=new Scanner(System.in);
        
        out.println("Enter Source File Name:");
        String fileName=Sc.nextLine();
        
        fileName+=".txt";
        
        File srcFile=new File(fileName);
        
        if(!srcFile.exists()){
        out.println("File doesnt exists");
        return;}
        
        out.println("1.Print Character by Character\n2.PRint the entire File\n3.Copy contents of one File to the other\n");
        int ch=Sc.nextInt();
        
        switch(ch){
            case 1:printCharacter(srcFile);
                    break;
            case 2:printFile(srcFile);
                    break;
            case 3:copyFileContent(srcFile);
                    break;
            default:out.println("Invalid Choice");
            
        }
    }
}
